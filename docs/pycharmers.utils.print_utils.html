Examples

>>> from pycharmers.utils import print_dict_tree
>>> print_dict_tree({"a": 0, "b": 1})

- b: 1
>>> print_dict_tree({"a": 0, "b": {"b1": 1, "b2": 2}})
- a: 0
- b:
  * b1: 1
  * b2: 2
>>> print_dict_tree({"a": 0, "b": {"b1": 1, "b2": {"b21": 0, "b22": 1}}, "c": 3})
- a: 0
- b:
  * b1: 1
  * b2:
    # b21: 0
    # b22: 1
- c: 3
